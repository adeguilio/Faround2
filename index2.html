  <script>

    const form = document.getElementById('rsvpForm');

    const attendanceField = document.getElementById('attendance');

    const ok = document.getElementById('success');

    const err = document.getElementById('error');


    function setRSVP(val){

      attendanceField.value = val;

      // small visual feedback

      ok.style.display = 'none'; err.style.display = 'none';

    }


    form.addEventListener('submit', async (e) => {

      // Ajax submit for nice UX; fallback is normal POST if this fails

      e.preventDefault();

      ok.style.display = 'none'; err.style.display = 'none';


      if(!attendanceField.value){

        attendanceField.value = 'Yes'; // default to Yes if user didn’t press a button

      }


      const data = new FormData(form);

      try{

        const res = await fetch(form.action, {

          method: 'POST',

          body: data,

          headers: { 'Accept': 'application/json' }

        });

        if (res.ok){

          form.reset();

          attendanceField.value = '';

          ok.style.display = 'block';

          window.scrollTo({ top: 0, behavior: 'smooth' });

        } else {

          throw new Error('Network response not ok');

        }

      } catch(e){

        err.style.display = 'block';

      }

    });

  </script>
